Controller:

            
    .controller('mainController', function($scope) {
        "use strict";
        $scope.dstest1={value:[150,50,250,30,120,90,200,50,0]};
        $scope.ds2={value:[50,90,0]};
        $scope.ds3={value:[250]};
    });
